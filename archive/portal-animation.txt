// Animation Configuration
const animationConfig = {
    soundUrl: 'https://cdn.shopify.com/s/files/1/0242/0175/6777/files/mixkit-futuristic-cinematic-noise-whip-1518.wav?v=1748880406',
    destinationUrl: 'https://your-destination-url.com', // Change this
    primaryColor: 0xff5011, // Orange color
    tunnelDuration: 2.5, // seconds
    redirectDelay: 4500 // ms total from click to redirect
};

// Audio setup
const cinematicWhip = new Audio(animationConfig.soundUrl);
cinematicWhip.preload = 'auto';
cinematicWhip.load();

// Three.js variables
let animationId = null;
let renderer, scene, camera, tube, stars;

// Audio fade helper
function fadeAudio(audioElement, targetVolume, duration) {
    const initialVolume = audioElement.volume;
    const delta = targetVolume - initialVolume;
    const startTime = Date.now();
    
    function updateVolume() {
        const elapsed = Date.now() - startTime;
        const progress = Math.min(elapsed / duration, 1);
        audioElement.volume = initialVolume + delta * progress;
        
        if (progress < 1) {
            requestAnimationFrame(updateVolume);
        }
    }
    
    updateVolume();
}

// Main animation trigger function
function startPortalAnimation(triggerElement, cardElement, logoImageUrl) {
    // Play sound
    cinematicWhip.currentTime = 0;
    cinematicWhip.volume = 1.0;
    cinematicWhip.play();
    
    // Hide trigger element/card
    if (cardElement) {
        cardElement.classList.add("fadeOut");
        setTimeout(() => {
            cardElement.style.display = "none";
        }, 500);
    }
    
    // Show tunnel canvas
    const canvasTunnel = document.getElementById("tunnelCanvas");
    canvasTunnel.style.display = "block";
    
    // Initialize Three.js tunnel
    initTunnel(animationConfig.primaryColor);
    
    // Create overlay elements
    const darkOverlay = document.createElement("div");
    darkOverlay.className = "dark-overlay";
    document.body.appendChild(darkOverlay);
    
    const logoOverlay = document.createElement("div");
    logoOverlay.className = "logo-overlay";
    if (logoImageUrl) {
        logoOverlay.style.backgroundImage = `url('${logoImageUrl}')`;
    }
    document.body.appendChild(logoOverlay);
    
    // Animation timeline
    setTimeout(() => {
        darkOverlay.style.opacity = "0.1";
        
        const darkFadeInterval = setInterval(() => {
            const currentOpacity = parseFloat(darkOverlay.style.opacity || "0");
            if (currentOpacity < 1) {
                darkOverlay.style.opacity = (currentOpacity + 0.05).toString();
            } else {
                clearInterval(darkFadeInterval);
            }
        }, 50);
        
        // Show logo at 1.5s
        setTimeout(() => {
            logoOverlay.classList.add("visible");
        }, 500);
        
        // Hide tunnel at 3.0s
        setTimeout(() => {
            cleanupTunnel();
            darkOverlay.style.backgroundColor = "#0a0a0a";
        }, 2000);
        
        // Fade out sound at 3.5s
        setTimeout(() => {
            fadeAudio(cinematicWhip, 0, 500);
        }, 2500);
        
        // Fade out logo at 4.0s
        setTimeout(() => {
            logoOverlay.classList.add("fade-out");
        }, 3000);
        
        // Redirect at 4.5s
        setTimeout(() => {
            window.location.href = animationConfig.destinationUrl;
        }, 3500);
    }, 1000);
}

// Three.js Tunnel Initialization
function initTunnel(primaryColor) {
    const w = window.innerWidth;
    const h = window.innerHeight;
    
    // Renderer with transparent background
    renderer = new THREE.WebGLRenderer({
        canvas: document.getElementById("tunnelCanvas"),
        antialias: true,
        alpha: true
    });
    renderer.setSize(w, h);
    renderer.setClearColor(0x000000, 0);
    
    // Scene
    scene = new THREE.Scene();
    scene.fog = new THREE.FogExp2(0x000000, 0.005);
    
    // Camera
    camera = new THREE.PerspectiveCamera(60, w / h, 0.1, 1000);
    
    // Create path
    const path = createCircularPath();
    
    // Tube geometry
    const tubeGeometry = new THREE.TubeGeometry(
        path,
        200,
        3,
        12,
        false
    );
    
    // Tube material
    const material = new THREE.MeshBasicMaterial({
        side: THREE.BackSide,
        wireframe: true,
        color: primaryColor,
        transparent: true,
        opacity: 0.8
    });
    
    // Tube mesh
    tube = new THREE.Mesh(tubeGeometry, material);
    scene.add(tube);
    
    // Stars
    const starsGeometry = new THREE.BufferGeometry();
    const starsMaterial = new THREE.PointsMaterial({
        color: 0xffffff,
        size: 1,
        transparent: true
    });
    
    const starsPositions = new Float32Array(2000 * 3);
    for (let i = 0; i < 2000; i++) {
        starsPositions[i * 3] = (Math.random() - 0.5) * 2000;
        starsPositions[i * 3 + 1] = (Math.random() - 0.5) * 2000;
        starsPositions[i * 3 + 2] = (Math.random() - 0.5) * 2000;
    }
    
    starsGeometry.setAttribute('position', new THREE.BufferAttribute(starsPositions, 3));
    stars = new THREE.Points(starsGeometry, starsMaterial);
    scene.add(stars);
    
    // Animation
    let pct = 0;
    const duration = 2.5;
    const startTime = Date.now();
    
    function animate() {
        const elapsed = (Date.now() - startTime) / 1000;
        pct = Math.min(elapsed / duration, 1);
        
        const lookAheadPct = Math.min(pct + 0.01, 1);
        const pt1 = path.getPointAt(pct);
        const pt2 = path.getPointAt(lookAheadPct);
        
        camera.position.set(pt1.x, pt1.y, pt1.z);
        camera.lookAt(pt2);
        
        renderer.render(scene, camera);
        
        if (pct < 1) {
            animationId = requestAnimationFrame(animate);
        }
    }
    
    animate();
    
    // Handle resize
    window.addEventListener('resize', () => {
        const w = window.innerWidth;
        const h = window.innerHeight;
        camera.aspect = w / h;
        camera.updateProjectionMatrix();
        renderer.setSize(w, h);
    });
}

// Create circular path
function createCircularPath() {
    const curve = new THREE.CatmullRomCurve3([
        new THREE.Vector3(0, 0, 0),
        new THREE.Vector3(20, 10, -50),
        new THREE.Vector3(40, -10, -100),
        new THREE.Vector3(60, 15, -150),
        new THREE.Vector3(50, -5, -200),
        new THREE.Vector3(0, 0, -250),
        new THREE.Vector3(-100, 0, -200),
        new THREE.Vector3(-150, 0, -100),
        new THREE.Vector3(-100, 0, 0),
        new THREE.Vector3(-50, 10, 100),
        new THREE.Vector3(-20, -10, 150),
        new THREE.Vector3(0, 0, 200)
    ]);
    
    curve.tension = 0.1;
    return curve;
}

// Cleanup Three.js
function cleanupTunnel() {
    if (animationId) {
        cancelAnimationFrame(animationId);
    }
    if (renderer) {
        renderer.dispose();
    }
    if (scene) {
        while(scene.children.length > 0){ 
            scene.remove(scene.children[0]); 
        }
    }
    const canvasTunnel = document.getElementById("tunnelCanvas");
    if (canvasTunnel) {
        canvasTunnel.style.display = "none";
    }
}

// Initialize animation on button click
document.addEventListener('DOMContentLoaded', function() {
    const button = document.getElementById('portalButton');
    const card = document.getElementById('portalCard');
    const logoUrl = 'your-logo.png'; // Replace with your logo URL
    
    button.addEventListener('click', function(e) {
        e.stopPropagation();
        startPortalAnimation(button, card, logoUrl);
    });
});
